		ai_will_do = { factor = 1 }
		available = {
			if = {
				limit = {
					OR = {

						is_subject_of = USA

					}
				}
				USA = { controls_state = 473 }
			}
			else = {
				USA = { surrender_progress > 0.15 }
			}
		}
		allowed = {
			OR = {
				TAG = ENG
				TAG = USA
				TAG = BCO
			}
		}
		selectable_mission = no
		is_good = no
		fire_only_once = no

		activation = {
	 		hidden_trigger = { has_country_flag = USA_operation_annex_british_columbia_flag }
		}
		timeout_effect = {
			hidden_effect = {
				every_country = {
					limit = {
						OR = {
							is_in_faction_with = USA
							is_subject_of = USA
							TAG = USA
						}
					}
					every_country = {
						limit = {
							OR = {
								is_in_faction_with = ENG
								is_subject_of = ENG
								TAG = ENG
							}
							has_war_with = PREV
						}
						white_peace = PREV
					}
				}
				clr_country_flag = USA_operation_annex_british_columbia_flag
			}
			if = {
				limit = {
					OR = {
						is_in_faction_with = USA
						is_subject_of = USA
						TAG = USA
					}
				}
				custom_effect_tooltip = operation_annex_british_columbia_usa_loses_tooltip
			}
			else = {
				custom_effect_tooltip = operation_annex_british_columbia_eng_wins_tooltip
			}
		}
		complete_effect = {
			hidden_effect = {
				every_country = {
					limit = {
						OR = {
							is_in_faction_with = USA
							is_subject_of = USA
							TAG = USA
						}
					}
					every_country = {
						limit = {
							OR = {
								is_in_faction_with = ENG
								is_subject_of = ENG
								TAG = ENG
							}
							has_war_with = PREV
						}
						white_peace = PREV
					}
				}
				clr_country_flag = USA_operation_annex_british_columbia_flag
			}
			if = {
				limit = {
					OR = {
						is_in_faction_with = USA
						is_subject_of = USA
						TAG = USA
					}
				}
				custom_effect_tooltip = operation_annex_british_columbia_usa_wins_tooltip
				USA = {
					if = {
						limit = { USA = { NOT = { owns_state = 473 } } }
						transfer_state = 473
					}
				}
			}
			else = {
				custom_effect_tooltip = operation_annex_british_columbia_eng_loses_tooltip
			}
		}
