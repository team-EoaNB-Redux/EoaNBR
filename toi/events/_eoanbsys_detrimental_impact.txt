	title = afr_rrp_racial_tension_increase.30.t

	is_triggered_only = yes
	hidden = yes
	immediate = {
		OWNER = { add_to_variable = { var = total_detrimental_colonial_impact_value value = 30 } }
		add_to_variable = { var = ROOT.detrimental_colonial_impact_value value = 30 }
		ROOT = {
			if = {
				limit = { OWNER = { NOT = { has_idea = afr_total_detrimental_colonial_impact_idea } } }
				OWNER = { add_ideas = afr_total_detrimental_colonial_impact_idea }
			}
		}
		if = {
			limit = { OWNER = { is_subject = yes } }
			OWNER = { OVERLORD = { add_to_variable = { var = ROOT.detrimental_colonial_impact_value@THIS value = 30 } } }
		}
		else = {
			OWNER = { add_to_variable = { var = ROOT.detrimental_colonial_impact_value@THIS value = 30 } }
		}
	}
}
