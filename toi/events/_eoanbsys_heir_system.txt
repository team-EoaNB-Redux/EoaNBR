#Weak heir gets dethroned
country_event = {
	id = heir.4
	title = heir.4.t
	desc = heir.4.d
	picture = GFX_report_event_saf_coup

	immediate = {
		# log = "[THIS.GetTag]: firing = heir.4"
	}

	trigger = {
		has_idea = weak_heir
		has_stability < 0.5
	}

	mean_time_to_happen = {
		days = 3650
		#modifier = {
		#	factor = 0.75
		#	has_stability < 0.4
		#}
		#modifier = {
		#	factor = 0.7
		#	has_stability < 0.3
		#}
		#modifier = {
		#	factor = 0.65
		#	has_stability < 0.2
		#}
	}

	immediate = {
		# log = "[THIS.GetTag]: firing = heir.4"
		# log = "[THIS.id] firing heir.4"
	}

	option = {
		name = heir.4.a
		# log = "option:		name = heir.4.a"
		ai_chance = { factor = 70 }
		swap_ideas = { remove_idea = weak_heir add_idea = dethroned_heir }
	}

	option = {
		name = heir.4.b
		# log = "option:		name = heir.4.b"
		ai_chance = { factor = 30 }
		add_political_power = -50
		swap_ideas = { remove_idea = weak_heir add_idea = no_heir }
	}
}
